 let searchArray = [1,2,3,4,5,6];



function randomSearch(searchArray,startNum = 0,endNum = searchArray.length, target) {


    let max = startNum;
    let min = endNum;

    let randomNum = Math.floor(Math.random() * (max - min) + min);

    if (target == searchArray[randomNum]) {
        return randomNum;
    } else if (target > searchArray[randomNum]) {
        return randomSearch(searchArray, randomNum + 1, max, target);
    } else if(target < searchArray[randomNum]){
        return randomSearch(searchArray, min, randomNum, target);
    } else {
        return -1;
    }



            }
